## 算法
### 实现栈

    class Stack {
        private items: any[];
        constructor(){
            this.items = [];
        }
        push(...elements:any[]){
            this.items = this.items.concat(elements);
        }
        pop():any{
            return this.items.pop();
        }
        isEmpty():boolean{
            return this.items.length === 0;
        }
        clear(){
            this.items = [];
        }
        size():number{
            return this.items.length;
        }
        print():string{
            return this.items.toString();
        }
    }

### 队列

     class Queue{
        private items: any[];
        constructor(){
            this.items = [];
        }
        enqueue(...elements:any[]){
            this.items = this.items.concat(elements);
        }
        dequeue(): any{
            return this.items.shift();
        }
        front():any{
            return this.items[0];
        }
        clear(){
            this.items = [];
        }
        isEmpty():boolean{
            return this.items.length === 0;
        }
        size():number{
            return this.items.length;
        }
        print(){
            console.log(this.items.toString());
        }
    }
